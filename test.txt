                            SELECT 
                                ep.post, 
                                ep.title, 
                                ep.time, 
                                c.email
                            FROM(
                                SELECT id AS feed_id, name AS feed_name
                                FROM users AS u
                                WHERE id = ?
                                OR
                                 id IN(SELECT friend_id FROM friends AS f WHERE f.user_id = ?)
                                 OR
                                   id IN(SELECT enemy_id FROM enemies AS e WHERE e.user_id = ?)
                                    
                                )AS feed
                                WHERE feed.name IN (
                                SELECT u.name, e.name
                                FROM users AS u
                                JOIN enemies AS e ON u.id = e.user_id
                                JOIN users AS ue on ue.id = e.enemy_id
                                WHERE u.id = ?
                                )
                                OR feed.name IN(
                                SELECT e.name
                                FROM enemies
                                JOIN users AS u ON u.id = e.user_id
                            JOIN enemy_posts AS ep ON ep.user_id = feed.feed_id
                            JOIN credentials AS c ON c.user_id = feed.feed_id
                            GROUP BY ep.post, ep.title, ep.time, c.email
                            ORDER BY ep.time DESC
